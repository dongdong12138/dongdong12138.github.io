# 插槽

## 默认插槽

```html title="子组件"
<template>
  <div>
    <slot></slot>
  </div>
</template>
```

```html title="父组件"
<template>
  <Dialog>
    
    <template v-slot:default>
      <div>123</div>
    </template>
    
    <!-- 或者 -->
    <template v-slot>
      <div>123</div>
    </template>

    <!-- 或者 -->
    <div>123</div>
    
  </Dialog>
</template>
```

## 具名插槽

```html title="子组件"
<template>
  <div>
    <slot name="header"></slot>
    <slot></slot>
    <slot name="footer"></slot>
  </div>
</template>
```

```html title="父组件"
<template>
  <Dialog>
    <template v-slot:header>
      <div>123</div>
    </template>
    <template v-slot>
      <div>456</div>
    </template>
    <template v-slot:footer>
      <div>789</div>
    </template>
  </Dialog>

  <!-- 或者也可以简写 -->
  <Dialog>
    <template #header>
      <div>123</div>
    </template>
    <template #default>
      <div>456</div>
    </template>
    <template #footer>
      <div>789</div>
    </template>
  </Dialog>
</template>
```

## 作用域插槽

```html title="子组件"
<div>
  <slot name="header"></slot>
  <div>
    <div v-for="item in 100">
      <slot :data="item"></slot>
    </div>
  </div>
  <slot name="footer"></slot>
</div>
```

```html title="父组件"
<Dialog>
  <template #header>
    <div>123</div>
  </template>
  <template #default="{ data }">
    <div>{{ data }}</div>
  </template>
  <template #footer>
    <div>789</div>
  </template>
</Dialog>
```

## 动态插槽

```html
<script setup>
const name = ref('header')
</script>

<template>
  <Dialog>
    <template #[name]>
      <div>123</div>
    </template>
  </Dialog>
</template>
```
